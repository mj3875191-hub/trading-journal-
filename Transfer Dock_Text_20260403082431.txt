auth.onAuthStateChanged(user => {

    document.getElementById("journal-section").style.display = "block";

function register(){

  const password = document.getElementById("password").value;
  auth.createUserWithEmailAndPassword(email, password)
      .catch(e => alert(e.message));

function login(){

  const password = document.getElementById("password").value;
  auth.signInWithEmailAndPassword(email, password)
      .catch(e => alert(e.message));

function googleLogin(){

  auth.signInWithPopup(provider).catch(e => alert(e.message));

function logout(){ auth.signOut(); }

// 🔹 Journal Functions
function addTrade(){
  const pair = document.getElementById("pair").value;
  const entry = parseFloat(document.getElementById("entry").value);
  const exit = parseFloat(document.getElementById("exit").value);
  const lot = parseFloat(document.getElementById("lot").value);
  if(!pair || !entry || !exit || !lot){ alert("Fill all fields"); return; }

  const profitLoss = (exit - entry) * lot; // simple calc
  db.collection("users").doc(auth.currentUser.uid).collection("trades")
    .add({pair, entry, exit, lot, profitLoss, timestamp: Date.now()})
    .then(() => loadTrades());
}

function loadTrades(){
  const table = document.getElementById("tradeTable");
  table.innerHTML = "<tr><th>Pair</th><th>Entry</th><th>Exit</th><th>Lot</th><th>Profit/Loss</th></tr>";
  let total = 0, wins = 0, totalTrades = 0, maxDD = 0, profitLossSum = 0;
  db.collection("users").doc(auth.currentUser.uid).collection("trades")
    .orderBy("timestamp")
    .get()
    .then(snapshot => {
      snapshot.forEach(doc => {
        const t = doc.data();
        const row = table.insertRow();
        row.insertCell(0).innerText = t.pair;
        row.insertCell(1).innerText = t.entry;
        row.insertCell(2).innerText = t.exit;
        row.insertCell(3).innerText = t.lot;
        row.insertCell(4).innerText =t.profitLoss.toFixed(2);

        totalTrades++;
        profitLossSum +=t.profitLoss;
        if(t.profitLoss>0) wins++;
        if(t.profitLoss<maxDD) maxDD =t.profitLoss;
      });
      document.getElementById("totalTrades").innerText = totalTrades;
      document.getElementById("winRate").innerText = totalTrades ? ((wins/totalTrades)*100).toFixed(2) : 0;
      document.getElementById("profitLoss").innerText = profitLossSum.toFixed(2);
      document.getElementById("drawdown").innerText = maxDD.toFixed(2);
    });
}
